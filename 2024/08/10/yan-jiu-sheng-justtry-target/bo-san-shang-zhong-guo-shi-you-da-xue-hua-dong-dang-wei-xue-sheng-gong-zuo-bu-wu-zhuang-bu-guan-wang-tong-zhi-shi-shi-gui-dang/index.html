        # 如果未找到任何元素
        assert title_spans, f'{url} 未找到具有类名 "Article_Title" 的 <span> 元素。'

        for index, title_span in enumerate(title_spans):
            a_tag = title_span.find('a')  # 找到 <span> 内的 <a> 标签
          
            ### 文章链接
            href = a_tag['href']  # 获取 href 属性
            if href[0]=='/':
                href = basic_url+href
            print(href)
          
            ### 日期
            # Date = href[1:5]+'-'+href[6:8]+'-'+href[8:10] # href[0]=='/'

            Date = PublishDate_spans[index].get_text()
            #
          
            ### 文章名称
            article_title = title_span.get_text()
            #
            # breakpoint()
            string = Date+': ['+article_title+']'+'('+href+')'
            article_list.append(string)
            # print(f'文章标题 {index + 1}: {article_title}')

            pass

        pass
